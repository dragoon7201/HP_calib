from epics import PV
import time
import winsound
import os
RF = PV("ACCL:RF:on")

while True:
    if RF.get(as_numpy=True) == 1:
        time.sleep(10)
    else:
        winsound.PlaySound("SystemExit", winsound.SND_ALIAS)
        print("!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!RF TRIP!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!\n"*10)
        input()
        os.system('cls')
